199 void modbus_handshake() {
200
201 switch (connection_state) {
202 case disconnected:
203 if (UART.get_queue_size() == 0 && has_pause_timer_expired()) {
204 is_paused = false;
205 new_data(); // clear nenw data flag
206 num_discovery_pings_received = 0;
207 enqueue_ping_msg();
208
209 connection_state = discovery;
210 } else {
211 // run_in() should be pulling any residual messages off the queue as they are received or timeout.
212 }
213 break;
214
215 case discovery:
216
217 // Note that the run_in() function should claim any responses that are ready,
218 // set a new_data flag and save the transaction to the response member
219 if (new_data()) {
220 if (response->is_echo_response() && response->is_reception_valid()) {
221
222 num_discovery_pings_received++;
223
224 if (num_discovery_pings_received >= connection_config.req_num_discovery_pings) {
225 synchronize_memory_map(); // loads many read messages onto the queue
226 connection_state = synchronization;
227 } else {
228 enqueue_ping_msg();
229 }
230 }
231 // new response was failed, or the wrong kind of response
232 else {
233 disconnect();
234 }
235 }
236 break;
237
238
239 case synchronization:
240
241 if (new_data() && !response->is_reception_valid()) {
242 // this should allow all the crap we put on the queue to timeout or be received before reattempting a connection
243 disconnect();
244 }
245 else {
246 if (UART.get_queue_size() == 0) {
247 enqueue_change_connection_status_fn(
248 connection_config.server_address,
249 true,
250 connection_config.target_baud_rate_bps,
251 connection_config.target_delay_us);
252 connection_state = negotiation;
253 }
254 }
255 break;
256
257
258 case negotiation:
259
260 if (new_data()) {
261
262 // Server responded to our change connection request with its realized baud and delay
263 if (response->get_rx_function_code() == change_connection_status && response->is_reception_valid()) {
264 uint8_t* rx_data = response->get_rx_data();
265 UART.adjust_baud_rate(
266 (uint32_t(rx_data[2]) << 24)
267 | (uint32_t(rx_data[3]) << 16)
268 | (uint32_t(rx_data[4]) << 8)
269 | (uint32_t(rx_data[5]) << 0)); //set baud
270
271 UART.adjust_interframe_delay_us(
272
273 (uint16_t(rx_data[6]) << 8) | rx_data[7]); //set delay
274
275 // Reduce timeouts
276 UART.adjust_response_timeout(connection_config.response_timeout_us);
277
278 connection_state = connected;
279
280 }
281 // Server failed to respond to our change connection request
282 else {
283 disconnect();
284 }
285 }
286 case connected:
287 // we already won!
288 break;
289 }
290 }

376 int enqueue_change_connection_status_fn(uint8_t device_address, bool connect, uint32_t baud_rate_bps, uint16_t delay_us) {
377 Transaction transaction;
378 uint16_t requested_state;
379 connect ? requested_state = 0xFF00 : requested_state = 0;
380
381 uint8_t data[8] = { uint8_t(requested_state >> 8),
382 uint8_t(requested_state),
383 uint8_t(baud_rate_bps >> 24),
384 uint8_t(baud_rate_bps >> 16),
385 uint8_t(baud_rate_bps >> 8),
386 uint8_t(baud_rate_bps),
387 uint8_t(delay_us >> 8),
388 uint8_t(delay_us) };
389
390 transaction.load_transmission_data(device_address, change_connection_status, data, 8, get_app_reception_length(change_connection_status));
391 int check = UART.enqueue_transaction(transaction);
392 return check;
