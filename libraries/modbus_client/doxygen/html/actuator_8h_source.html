182 void run_in() {
183
184
185 modbus_client.run_in();
186
187 if ( modbus_client.is_response_ready() ) {
188 response = modbus_client.dequeue_transaction();
189 new_data_flag = true; // communicate to other layers that new data was received
190
191 if ( !response->is_reception_valid() ) {
192 cur_consec_failed_msgs++;
193 failed_msg_counter++;
194 if(connection_state == connected && cur_consec_failed_msgs >= connection_config.max_consec_failed_msgs){
195 disconnect();
196 }
197 }
198
199 // Response was valid
200 else {
201
202 cur_consec_failed_msgs = 0;
203 success_msg_counter++;
204
205 switch (response->get_rx_function_code()) {
206
207 case read_holding_registers:{
208 // add the received data to the local copy of the memory map
209 u16 register_start_address = (response->get_tx_data()[0] << 8) + response->get_tx_data()[1];
210 u16 num_registers = (response->get_tx_data()[2] << 8) + response->get_tx_data()[3];
211 for ( int i = 0; i < num_registers; i++ ) {
212 u16 register_data = (response->get_rx_data()[1 + i*2] << 8) + response->get_rx_data()[2 + i*2];
213 orca_reg_contents[register_start_address + i] = register_data;
214 }
215 break;
216 }
217 case write_single_register:
218 // nothing to do
219 break;
220
221 case motor_command:
222 orca_reg_contents[POS_REG_H_OFFSET] = (response->get_rx_data()[ 0] << 8) | response->get_rx_data()[ 1];
223 orca_reg_contents[POS_REG_OFFSET] = (response->get_rx_data()[ 2] << 8) | response->get_rx_data()[ 3];
224 orca_reg_contents[FORCE_REG_H_OFFSET] = (response->get_rx_data()[ 4] << 8) | response->get_rx_data()[ 5];
225 orca_reg_contents[FORCE_REG_OFFSET] = (response->get_rx_data()[ 6] << 8) | response->get_rx_data()[ 7];
226 orca_reg_contents[POWER_REG_OFFSET] = (response->get_rx_data()[ 8] << 8) | response->get_rx_data()[ 9];
227 orca_reg_contents[TEMP_REG_OFFSET] = (response->get_rx_data()[10]);
228 orca_reg_contents[VOLTAGE_REG_OFFSET] = (response->get_rx_data()[11] << 8) | response->get_rx_data()[12];
229 orca_reg_contents[ERROR_REG_OFFSET] = (response->get_rx_data()[13] << 8) | response->get_rx_data()[14];
230 break;
231
232 case read_coils :
233 case read_discrete_inputs :
234 case read_input_registers :
235 case write_single_coil :
236 case read_exception_status :
237 case diagnostics :

577 void enqueue_motor_frame() {
578
579
580 if(modbus_client.get_queue_size() >= 2) return;
581
582 switch(comms_mode){
583 case SleepMode:
584 motor_command_fn(connection_config.server_address, 0, 0); //any register address other than force or position register_adresses will induce sleep mode and provided register_value will be ignored
585 break;
586 case ForceMode: {
587 if(uint32_t(modbus_client.get_system_cycles() - stream_timeout_start) > stream_timeout_cycles){ //return to sleep mode if stream timed out
588 comms_mode = SleepMode;
589 }
590 else {
591 motor_command_fn(connection_config.server_address, FORCE_CMD, force_command);
592 }
593 break;
594 }
595 case PositionMode:
596 if(!POS_CTRL){
597 return;
598 }
599
600 if(uint32_t(modbus_client.get_system_cycles() - stream_timeout_start) > stream_timeout_cycles){ //return to sleep mode if stream timed out
601 comms_mode = SleepMode;
602 }
603 else {
604 motor_command_fn(connection_config.server_address, POS_CMD, position_command);
605 }
606 break;
607 }
