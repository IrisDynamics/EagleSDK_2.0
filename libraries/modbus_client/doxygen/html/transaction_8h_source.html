119 void load_transmission_data(uint8_t address, uint8_t function_code, uint8_t *data, int num_data, int num_expected_rx){
120 set_ID();
121 tx_buffer_size = 4 + num_data; //1 address byte + 1 function code byte + 2 CRC bytes = 4 bytes
122 tx_buffer_index = 0;
123 tx_buffer[0] = address;
124 tx_buffer[1] = function_code;
125 int i;
126 for(i = 2; i < num_data + 2; i++) tx_buffer[i] = data[i - 2];
127 uint16_t crc = ModbusCRC::generate(tx_buffer, tx_buffer_size - 2);
128 tx_buffer[i++] = uint8_t(crc >> 8);
129 tx_buffer[i] = uint8_t(crc);
130 reception_length = num_expected_rx;
131 }
